function caseInSwitch(val) {
  let answer = "";
  // Only change code below this line
  switch(val) {
  case 1:
    val === 1;
    answer = "alpha";
    break;
  case 2:
    val === 2;
    answer = "beta";
    break;
  case 3:
    val === 3;
    answer = "gamma";
    break;
  case 4:
    val === 4;
    answer = "delta";
    break;
}


  // Only change code above this line
  return answer;
}

caseInSwitch(1);
